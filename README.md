## for loop first basic program in C

#include <stdio.h>

int main() {
    
    for (int i = 1; i <= 5; i++) 
    
    {
        printf("%d\n", i);
    }
    
    return 0;
}

## Output
    1
    2
    3
    4
    5


## While loop first basic program in C

#include <stdio.h>

int main() {

    int i = 1; 
    while (i <= 5) {
        printf("Number: %d\n", i);
        i++; 
    }

    return 0;
}    
## Output

    Number: 1
    Number: 2
    Number: 3
    Number: 4
    Number: 5

## Do-While loop first basic program in C

#include <stdio.h>

int main() {
    
    int i = 1; 

    do {
        printf("Number: %d\n", i);
        i++; 
    } while (i <= 5); 
    return 0;
}

## Output

    Number: 1
    Number: 2
    Number: 3
    Number: 4
    Number: 5
